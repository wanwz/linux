## 多维数组遍历
```go
package main

import "fmt"

func main() {
    var arys [2][3]int = [...][3]int{{1,2,3}, {4,5,6}}
    for k1, v1 := range arys {
        for k2, v2 := range v1 {
            fmt.Printf("(%d,%d)=%d ", k1, k2, v2)
        }
        fmt.Println()
    }
}

/*
结果：
(0,0)=1 (0,1)=2 (0,3)=3
(1,0)=4 (1,1)=5 (1,3)=6
*/
```

## 数组拷贝和传参
```go
package main
import "fmt"
func printArr(arr *[5]int) {
    arr[0] = 10
    for i, v := range arr {
        fmt.Println(i, v)
    }
}

func main() {
    var arr1 [5]int
    printArr(&arr1)
    fmt.Println(arr1)
    arr2 := [...]int{2, 4, 6, 8, 10}
    printArr(&arr2)
    fmt.Println(arr2)
}
```
